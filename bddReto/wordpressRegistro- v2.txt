<?php
session_start();
echo "Usuario".$_SESSION["usuario"];


if(isset($_POST["enviLog"])){  //Nombre del boton del html
    
    $server = 'localhost';
    $username = 'user_wordpress';
    $password = 'pass_wordpress';
    $database = 'wordpress';

    try {
        $base = new PDO("mysql:host=$server;dbname=$database;", $username, $password);
        $base->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);

        $message = '';  //El dato de message se rellena con la variable de abajo si se ha creado el Usuario o no
        
            $sql = "INSERT INTO usuarios (vchEmail, vchNombreUser, vchPassword, vchRol) VALUES (:email, :username,
            :password, :rol)";
            $stmt = $base->prepare($sql);
            $stmt->bindValue(':email', $_POST['email']);
            $stmt->bindValue(':username', $_POST['username']);

            $contra= $_POST['password'];
            $contrasenaHash= password_hash($contra, PASSWORD_DEFAULT);

            $stmt->bindValue(':password', $contrasenaHash);


            
            $stmt->bindValue(':rol', $_POST['rol']);
            
            if ($stmt->execute()) {
                $message = 'Successfully created new user';
                echo "$message";
            } else {
                $message = 'Sorry there must have been an issue creating your account';
                echo "$message";
            }
      
        
    } catch (PDOException $e) {
        die('Connection Failed: ' . $e->getMessage());
    } //catch
} //isset
?>
